import React, { useState, useEffect, useCallback } from 'react'; /** * App: GlauWeb Instant Saver * Função: Detetar HTML no clipboard e salvar como ficheiro automaticamente. * Versão: Resiliente a bloqueios de segurança do navegador. */ const App = () => { const [status, setStatus] = useState('A iniciar...'); const [showButton, setShowButton] = useState(false); // Função para processar o texto e disparar o download const handleHTMLContent = useCallback((text) => { if (!text || text.trim() === '') { setStatus('Conteúdo vazio.'); return; } // Verifica se o conteúdo parece ser HTML if (!text.includes('<') || !text.includes('>')) { setStatus('Conteúdo não parece ser HTML válido.'); return; } // Tenta extrair o título do HTML let fileName = 'index'; const titleMatch = text.match(/(.*?)<\/title>/i) || text.match(/<h1>(.*?)<\/h1>/i); if (titleMatch && titleMatch[1]) { fileName = titleMatch[1] .trim() .toLowerCase() .normalize("NFD").replace(/[\u0300-\u036f]/g, "") .replace(/[^a-z0-9]/gi, '_') .substring(0, 50); } if (!fileName) fileName = 'index'; try { const blob = new Blob([text], { type: 'text/html;charset=utf-8' }); const url = URL.createObjectURL(blob); const link = document.createElement('a'); link.href = url; link.download = `${fileName}.html`; document.body.appendChild(link); link.click(); document.body.removeChild(link); URL.revokeObjectURL(url); setStatus(`Salvo: ${fileName}.html ✅`); } catch (err) { setStatus('Erro ao gerar o ficheiro.'); } }, []); // Tenta ler via API (pode falhar por política de segurança) const tryClipboardRead = async () => { try { setStatus('A tentar ler automaticamente...'); const text = await navigator.clipboard.readText(); handleHTMLContent(text); } catch (err) { // Se falhar, pedimos interação manual setStatus('Aguardando colagem... (Pressione Ctrl+V)'); setShowButton(true); } }; // Efeito para capturar o "Paste" (Colar) manual e tentar o automático useEffect(() => { const handleGlobalPaste = (e) => { const pastedData = e.clipboardData.getData('text'); setStatus('Conteúdo colado detetado!'); handleHTMLContent(pastedData); }; // Adiciona o listener de colagem global window.addEventListener('paste', handleGlobalPaste); // Tenta ler automaticamente após um curto delay const timer = setTimeout(() => { tryClipboardRead(); }, 800); return () => { window.removeEventListener('paste', handleGlobalPaste); clearTimeout(timer); }; }, [handleHTMLContent]); return ( <div className="min-h-screen bg-[#F7F4E9] flex items-center justify-center p-6 font-sans"> <div className="max-w-sm w-full bg-white rounded-3xl shadow-xl border-2 border-[#E4BAD4] p-10 text-center"> <div className="mb-6 inline-flex items-center justify-center w-20 h-20 bg-[#C490D1] text-white rounded-2xl shadow-lg animate-pulse"> <svg xmlns="http://www.w3.org/2000/svg" className="h-10 w-10" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth={2}> <path strokeLinecap="round" strokeLinejoin="round" d="M13 10V3L4 14h7v7l9-11h-7z" /> </svg> </div> <h1 className="text-2xl font-black text-[#2D2830] mb-2"> GlauWeb <span className="text-[#C490D1]">Instant</span> </h1> <p className="text-sm text-[#756D79] mb-8 font-semibold"> Se o download não iniciar sozinho, <span className="text-[#C490D1]">clique no botão</span> ou pressione <span className="text-[#C490D1]">Ctrl+V</span>. </p> <div className="bg-[#F6DFEB] rounded-xl p-4 mb-8"> <p className="text-xs font-bold text-[#C490D1] uppercase tracking-widest mb-1">Status do Sistema</p> <p className="text-[#2D2830] font-bold truncate">{status}</p> </div> {showButton && ( <button onClick={tryClipboardRead} className="w-full bg-[#C490D1] hover:bg-[#b37dc1] text-white font-black py-4 px-6 rounded-2xl shadow-lg transition-all active:scale-95 flex items-center justify-center gap-3" > <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5" viewBox="0 0 20 20" fill="currentColor"> <path fillRule="evenodd" d="M4 4a2 2 0 012-2h4.586A2 2 0 0112 2.586L15.414 6A2 2 0 0116 7.414V16a2 2 0 01-2 2H6a2 2 0 01-2-2V4z" clipRule="evenodd" /> </svg> TENTAR COLAR AGORA </button> )} <div className="mt-10 pt-6 border-t border-gray-100 opacity-50"> <p className="text-[10px] font-black text-[#E4BAD4] uppercase tracking-tighter"> Prof. Glauber Santiago — UFSCar </p> </div> </div> </div> ); }; export default App;
